focus_tree = {
	id = MOD_	# позывной ветки (совет: тэг страны + focus + tree (пример: AUS_focus_tree) english please!) 

	country = {
		factor = 0 
		modifier = {
			add = 10 	# если пишите на ту страну где уде есть уникальная ветка то пишите 10 или увеличивайте до тех пор пока ваша не заменит ванильную
			original_tag = AUS 	# на какую страну ветка
		}
	}

	continuous_focus_position = {x = 100 y = 1500} 	# положение непрерывных фокусов
	
	initial_show_position = { 	# позиция камеры при первом заходе на вкладку "фокусы"
		x = 0 	# положение по горизонтали
		y = 0 	# положение по вертикали
	}

	default = no 	# если вратце заменяет стандартную, ветку непроработаных стран (всегда ставьте "no")

	focus = { 	# ключевое слово создания фокуса
		id =  MOD_ 	# позывной фокуса (совет: тэг страны + произвольное название (пример: AUS_STAN) english please!)
		icon = GFX_  # иконка фокуса (брать из корня игры gfx\interface\goals)
		cost = 10 	# число * 7 (10 = 70 дней изучения)

		prerequisite = {focus = MOD_} 	# фокус который открывает этот
		mutually_exclusive = {focus = MOD_} # взаимоисключающий фокус
		
		relative_position_id = MOD_  # относительная позиция от фокуса (то есть если 0, 0 изначально это позиция отнсительно правого врехнего угла, то так 0, 0 это уже смещение относительно выбранного фокуса)
		x = 0 	# положение фокуса по горизонтали
		y = 0 	# положение фокуса по вертикали

		available_if_capitulated = no 	# можно ли изучать после капитуляции
		cancel_if_invalid = yes 	# отменять если не саблюдаются требования (available)
		continue_if_invalid = no 	# продолжать изучать если не саблюдаются требования (если cancel_if_invalid и этот = no, то фокус при несоблюдении условий (available) встанет на паузу, до того времени, пока available опять будут соблюдаться)
		#обычно три строчки сверху не пишуться

		available = { 	# требования к началу изукения

		}

		bypass = { 	# пропустить фокус если...

		}

		completion_reward = { 	# эффекты после взятия фокуса

		}
	}
}
