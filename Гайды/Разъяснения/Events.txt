#значок "#" это заметка (коментарий если по правильному) для разроботичков, котрые не считаваются програмой, в нашем случае хойке. Можно удалить)

add_namespace = corse   # обязательная штука для ивентов, писать что угодно на английском (даже так course_test). 
add_namespace = course_news     # namespace может быть несколько!

country_event = {   # создание ивента
    id = course.1   # id как у фокусов, идей и т.д (первая часть всегда должна содержать в себе namespace)
    title = course.1.t      # ключ локализации для надписи сверху ивента
    desc = course.1.d   # ключ локализации для основного текста в ивенте

    is_triggered_only = yes     # чтобы не мог взываться сам
    fire_only_once = yes    # может быть вызван один раз (команды не считаются)
    picture = GFX_    # картинка в ивенте. Их можно найти в корне игры по пути "gfx\event_pictures"
        
    immediate = { # блок в котором записываются эффекты, которые будут выданы сразу при показе ивнта тому, кто видит его. 
        add_political_power = 100 # в данном случае сразу при появление ивента на экране, той стране у которой показался ивент дадут 100 политки
        FROM = { # благодаря этому блоку, всё что написано в нём будет применено к тому, кто инициировал (запустил) ивент. (FROM не только в этом блоке, в option тоже)
            add_political_power = 100
        }
    }

    option = { #1-ый вариант
        name = course.1.a   # ключ локализации для надписи первой кнопки 
        ai_chance = {factor = 50}   # шанс ИИ что он выберет именно тот вариант (можно модифицировать). Может быть больше 100 в сумме!

        create_wargoal = {  # создать цель войны
            type = annex_everything  #тип войны
            target = every_neighbor_country  # против кого (в нашем случае все соседние страны)
        }
    }   

    option = { #2-ой вариант
        name = course.1.b   # ключ локализации для надписи второй кнопки 
        ai_chance = {factor = 50    # первоночальный шанс выбора ИИ
            modifier = {    # модификатор
                add = 30    # сколько "очков шанса" добавить если...
                LUX = {     # Люксембруг...
                    has_government = communism # имеет правительство коммунизм
                }
            }
        }

        add_political_power = 100
        add_stability = 0.05  # эффект при выборе (никаких completion_reward, тупо в option)
    }
}

news_event = {  # новостной ивент
    id = course_news.1  # отдельный namespace кстати создавать необязательно, просто для удобства
    title = course_news.1.t
    desc = course_news.1.d

    is_triggered_only = yes
    picture = GFX_news_event_german_speech

    option = {  # в случае с ивентами это не отдельные кнопки, а разные надписи на кнопке при определённых тригерах. Поэтому в новостном ивенте обязаетлен triger чтобы понимать кому какой вариакнт показывать
        triger = {
            tag = LUX
        } # только Люксембургу будет показана новость с надписью "Великий Люксембруг" (ниже написана вся локализация)
        name = course_news.1.a
    }

    option = {
        triger = { # во втором чаще всего пишем обратный тригер, т.е тоже что и в первом только с блоком NOT (короче делаем взаимоисключающиеся)
            NOT = {
                tag = LUX
            }
        } # всем отсальным будет показана новость с надписью на кнопке "Люксембург был долго недооценён, и прежде всего из-за его размеров. Что предпримем?" (ниже написана вся локализация)
       name = course_news.1.b 
    }
}

#коментарии: name, tittle, desc мы будем прописывать в localization:

#l_russian:
# course.1.t: "Великий Люксембруг" - прописываем надпись сверху (она же будет во всплывающей подсказке: "произойдёт событие: ...")
# course.1.d: "Люксембург был долго недооценён, и прежде всего из-за его размеров. Что предпримем?" - надпись в ивенте
# course.1.a: "Расширение!" - надпись на первой кнопке
# course.1.b: "Нам не нужно чужое мнение" - надпись на второй кнопке

#буквы и их обозначения:
# t - title: надпись сверху нвости (заголовок), а также то, как будет называться новость во всплывающих подсказках
# d - desc - description - описание или же основной текст новости
# a, b, c, e - обозначение кнопок. Первая кнопка - a, вторая - b, третья - с, а вот четвёртая "e" из-за того что "d" занята описанием
